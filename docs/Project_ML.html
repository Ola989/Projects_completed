Predicting wine quality¶

Experiment uisng regression techniques to predict the quality of vinho verde white wine samples from Portugal wine using their physicochemical properties.

Dataset:

Wine Quality Dataset. Accessed May 18, 2021. https://archive.ics.uci.edu/ml/datasets/Wine+Quality

Prediction Techniques:

Support Vector Machine, Multiple Linear Regression, Decision Tree Regression

Objective:

Determine the best model for predicting the quality of vinho verde white wine from regression techniques using: all features from the dataset, feature selection by backward elimination, and feature extraction from dimensionality reduction using PCA and Kernel PCA.

Result:

The multiple linear regression model with backward eliminated features performed best using K-Fold model evaluation and Decision Tree regressor with Kernel PCA features performed least of all the models compared to predict quality

def model_2_way_split(a, X_train, X_test, y_train, y_test):
    m=[]
    i = 0
    while (i < len(a)):
        n =[]
        n.append(a[i][0])
        obj = a[i][1]
        obj.fit(X_train, y_train)
        y_pred = obj.predict(X_test)
        mse = mean_squared_error(y_test, y_pred)
        n.append(mse)
        m.append(n)
        i = i+1
    df = pd.DataFrame(m, columns=['Model','mean_squared_error'])
    return df.sort_values(by=['mean_squared_error'], ascending =True).reset_index(drop=True)

def backward_elimination(X, y, sL):
  X_b= np.append(arr=np.ones((len(X),1)).astype(int), values=X, axis=1) #add a column of ones to the features
  X_b = X_b.astype('float64') #convert to float data type
  r = list(np.arange(0, X_b.shape[1])) # create an array of the initial range to iterate over, using an interval that matches the number of columns in features-and-ones
  while(len(r)>0): # loop condition
    X_sig = X_b[:,r] #take all rows and columns as determined in the range, r
    obj_OLS = sm.OLS(endog = y, exog = X_sig).fit() #fit OLS
    p_value_Max = obj_OLS.pvalues.max() #get the maximum p-value
    if p_value_Max > sL: #condition to determine if p-value is greater than the chosen level of significance
      i = np.where(obj_OLS.pvalues == obj_OLS.pvalues.max())[0] #determine the position where p-value is maximum
      r.pop(int(i)) #eliminate this index from the list of indexes
    else:
      break
  r = r[1:] #take out the constant
  r = [a -1 for a in r] #correct for the indexes after the constant is removed
  print(obj_OLS.summary()) #print out summary results
  return r #return a list of indexes satisfying the condition

# automating k-fold cross validation for identified models using the given un/transformed data
def k_fold_cross_validate(a, X, y):
    l=[]
    m=[]
    k=[]
    i = 0
    while (i < len(a)):
        model_scores = cross_val_score(estimator= a[i][0], X=X, y=y, cv=5, scoring='neg_mean_absolute_error')
        m.append(a[i][1])
        k.append((a[i][1])+','+str(model_scores))
        l.append(model_scores.mean())
        i = i+1
    return l, m, k
# mean abs error made negative as it's a loss function. High (neg) value is best as it has the least loss.
